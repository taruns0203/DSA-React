// Create dummy node
const dummy = new Node(0);
dummy.next = head;
let groupPrev = dummy;
while (true) {
kthNode = getKth(groupPrev, k);
if (!kthNode) break;
groupStart = groupPrev.next;
groupNext = kthNode.next;
// Reverse k nodes
prev = groupNext;
curr = groupStart;
while (curr !== groupNext) {
next = curr.next;
curr.next = prev;
prev = curr; curr = next;
}
// Reconnect
groupPrev.next = prev;
groupPrev = groupStart;
}
return dummy.next;
