/** * Definition for singly-linked list. * function ListNode(val, next) { * this.val = (val===undefined ? 0 : val) * this.next = (next===undefined ? null : next) * } */ function addTwoNumbers(l1, l2) { // Dummy head to simplify list building const dummyHead = new ListNode(0); let current = dummyHead; let carry = 0; // Continue while there are digits to process or carry remains while (l1 !== null || l2 !== null || carry !== 0) { // Get digit values (0 if list ended) const val1 = l1 !== null ? l1.val : 0; const val2 = l2 !== null ? l2.val : 0; // Add digits plus carry const sum = val1 + val2 + carry; // Extract digit and new carry carry = Math.floor(sum / 10); const digit = sum % 10; // Create result node current.next = new ListNode(digit); current = current.next; // Move to next nodes if (l1 !== null) l1 = l1.next; if (l2 !== null) l2 = l2.next; } return dummyHead.next; }
