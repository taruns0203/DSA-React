function maxSubarray(nums) {
  let maxSum = -Infinity;
  for (let i = 0; i < n; i++) { // start index
    for (let j = i; j < n; j++) { // end index
      let sum = 0;
      for (let k = i; k <= j; k++) { // calculate sum
        sum += nums[k];
      }
      maxSum = Math.max(maxSum, sum);
    }
  }
  return maxSum;
}
