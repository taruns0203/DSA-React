function maxSubarray(nums) {
  let maxSum = -Infinity;
  for (let i = 0; i < n; i++) { // start index
    let currentSum = 0; // reset for each start
    for (let j = i; j < n; j++) { // end index
      currentSum += nums[j]; // O(1) - just add!
      // ‚ùå NO k loop needed!
      maxSum = Math.max(maxSum, currentSum);
    }
  }
  return maxSum;
}

üí° The Key Insight

Instead of recalculating the sum from scratch for each subarray, we extend the previous sum:

sum[i..j] = sum[i..j-1] + nums[j]

This simple change reduces each subarray's sum calculation from O(n) to O(1)!
