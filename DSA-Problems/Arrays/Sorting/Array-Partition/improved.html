📝 Algorithm
function arrayPairSum(nums) {
const count = new Array(21).fill(0);
for (const num of nums)
count[num + 10]++;
let sum = 0, needPair = false;
for (let v = -10; v <= 10; v++) {
const freq = count[v + 10];
if (freq === 0) continue;
if (!needPair)
sum += v * Math.ceil(freq/2);
else
sum += v * Math.floor(freq/2);
if (freq % 2 === 1) needPair = !needPair;
}
return sum;
}
