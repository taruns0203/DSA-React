📖 Algorithm
// Find tail & length
n=1; tail=head
while(tail.next) n++,tail=tail.next
k = k % n
if(k===0) return head
// Make circular
tail.next = head
// Walk n-k to newTail
newTail = head
for(i=1;i<n-k;i++) newTail=.next
newHead = newTail.next
// Break circle
newTail.next = null
return newHead
