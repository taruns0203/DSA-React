📖 Algorithm
let curr = head
while (curr) {
  if (curr.child) {
    // find tail of child
    tail = findTail(child)
    tail.next = curr.next
    curr.next.prev = tail
    curr.next = curr.child
    child.prev = curr
    curr.child = null
  }
  curr = curr.next
}
